```js:quick_sort.py
def	quick_sort(array, left, right):
	if left + 1 >= right:
		return

	pivot = array[right - 1]
	current_idx = left
	for i in range(left, right - 1):
		if array[i] < pivot:
			array[i], array[current_idx] = array[current_idx], array[i]
			current_idx += 1

	array[right - 1], array[current_idx] = array[current_idx], pivot

	quick_sort(array, left, current_idx)
	quick_sort(array, current_idx + 1, right)
```
